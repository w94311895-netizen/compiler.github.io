💡 Python Tips & Best Practices

Write cleaner, more efficient, and more professional Python code by following these proven best practices and tips.

1. Use Meaningful Variable Names

Bad:


a = 25
b = 30
c = a + b
print(c)

Good:


student_age = 25
class_count = 30
total = student_age + class_count
print(total)

2. Write Comments and Documentation


# Bad - no comments
def calc(x, y):
    return x * y + x / y

# Good - clear documentation
def calculate_student_score(test_score, assignment_score):
    """
    Calculate final student score based on test and assignment.
    
    Args:
        test_score: Score on the test (0-100)
        assignment_score: Score on assignment (0-100)
    
    Returns:
        Average of test and assignment scores
    """
    return (test_score + assignment_score) / 2

3. Follow PEP 8 Style Guide

PEP 8 is the official Python style guide. Key points:

  • Use 4 spaces for indentation (not tabs)
  • Maximum line length of 79 characters
  • Use lowercase with underscores for variables (snake_case)
  • Use UPPERCASE for constants
  • Use CamelCase for class names

# Good naming conventions
MAX_STUDENTS = 100
student_name = "John"
class_year = 2024

class StudentManager:
    def get_student_list(self):
        pass

4. Keep Functions Small and Focused

Bad:


def process_data(data):
    # Does too many things
    validated = []
    for item in data:
        if item > 0:
            validated.append(item * 2)
    result = sum(validated)
    print(f"Result: {result}")
    return result

Good:


def validate_positive(data):
    """Filter positive numbers"""
    return [item for item in data if item > 0]

def double_values(data):
    """Double all values"""
    return [item * 2 for item in data]

def process_data(data):
    """Main processing function"""
    validated = validate_positive(data)
    doubled = double_values(validated)
    return sum(doubled)

5. Use List Comprehensions

Before:


numbers = [1, 2, 3, 4, 5]
squared = []
for num in numbers:
    squared.append(num ** 2)

After:


numbers = [1, 2, 3, 4, 5]
squared = [num ** 2 for num in numbers]
# Or with condition
even_squared = [num ** 2 for num in numbers if num % 2 == 0]

6. Handle Exceptions Properly

Bad:


try:
    result = int(user_input) / 0
except:
    print("Error")

Good:


try:
    result = int(user_input) / 0
except ValueError:
    print("Invalid number format")
except ZeroDivisionError:
    print("Cannot divide by zero")
except Exception as e:
    print(f"Unexpected error: {e}")
finally:
    print("Cleanup code here")

7. Use Default Arguments Wisely


# Good for simple defaults
def greet(name="Guest"):
    return f"Hello, {name}!"

print(greet())           # Hello, Guest!
print(greet("Alice"))    # Hello, Alice!

# Avoid mutable defaults
# Bad
def add_item(item, list=[]):
    list.append(item)
    return list

# Good
def add_item(item, list=None):
    if list is None:
        list = []
    list.append(item)
    return list

8. Use Meaningful If/Else Conditions

Bad:


if user_age >= 18:
    is_adult = True
else:
    is_adult = False

Good:


is_adult = user_age >= 18

# Or for complex logic
if user_age >= 18 and user_has_license:
    can_drive = True

9. Use Built-in Functions


numbers = [3, 1, 4, 1, 5, 9, 2]

# Don't write loops for common tasks
print(len(numbers))          # Length
print(sum(numbers))          # Sum
print(min(numbers))          # Minimum
print(max(numbers))          # Maximum
print(sorted(numbers))       # Sorted
print(set(numbers))          # Unique values

10. Test Your Code


def calculate_discount(price, discount_percent):
    """Calculate price after discount"""
    if price < 0 or discount_percent < 0 or discount_percent > 100:
        raise ValueError("Invalid input")
    return price * (1 - discount_percent / 100)

# Test cases
try:
    assert calculate_discount(100, 10) == 90
    assert calculate_discount(50, 50) == 25
    print("All tests passed!")
except AssertionError:
    print("Test failed!")

11. Use Context Managers for Files

Bad:


file = open("data.txt")
data = file.read()
file.close()

Good:


with open("data.txt") as file:
    data = file.read()
# File automatically closes

12. Use Constants for Magic Numbers

Bad:


if age > 18 and height > 5.5:
    can_ride = True

Good:


MIN_AGE_REQUIREMENT = 18
MIN_HEIGHT_REQUIREMENT = 5.5

if age > MIN_AGE_REQUIREMENT and height > MIN_HEIGHT_REQUIREMENT:
    can_ride = True
